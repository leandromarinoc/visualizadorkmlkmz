function criarImagemCanvas(formato, cor = '#FFFFFF') {
    const size = 64;
    const canvas = document.createElement('canvas');
    canvas.width = size;
    canvas.height = size;
    const ctx = canvas.getContext('2d');
    ctx.strokeStyle = cor;
    ctx.lineWidth = 4;

    if (formato === 'triangulo') { // Transformador
        ctx.fillStyle = cor;
        ctx.beginPath();
        ctx.moveTo(size / 2, 10);
        ctx.lineTo(size - 10, size - 10);
        ctx.lineTo(10, size - 10);
        ctx.closePath();
        ctx.fill();
    } else if (formato === 'quadrado-nf') { // Chave Fechada
        ctx.fillStyle = cor;
        ctx.fillRect(12, 12, 40, 40);
    } else if (formato === 'quadrado-na') { // Chave Aberta
        ctx.strokeRect(12, 12, 40, 40);
    }

    return ctx.getImageData(0, 0, size, size);
}

    if (typeof VERSAO_ATUAL_SISTEMA !== 'undefined' && versaoNoCelular !== VERSAO_ATUAL_SISTEMA) {
        alert("🚀 Há uma nova versão disponível! O sistema será atualizado agora para a versão " + VERSAO_ATUAL_SISTEMA + ".");
        
        // Atualiza a versão no armazenamento
        localStorage.setItem('inspecao_versao_app', VERSAO_ATUAL_SISTEMA);
        
        // Limpa o cache do navegador para garantir que ele baixe o código novo
        if ('serviceWorker' in navigator) {
            caches.keys().then(names => {
                for (let name of names) {
                    caches.delete(name);
                }
            });
        }
        
        // Força o recarregamento da página ignorando o cache
        window.location.reload(true);
        
    } else {
        // Se já estiver na versão mais recente, funciona como um botão de "Reset/Limpar" padrão
        const confirmar = confirm("O sistema já está na versão mais recente (" + (typeof VERSAO_ATUAL_SISTEMA !== 'undefined' ? VERSAO_ATUAL_SISTEMA : 'Atual') + ").\n\nDeseja recarregar a página para limpar a tela e o mapa?");
        
        if (confirmar) {
            window.location.reload(true);
        }
    }
}

function salvarCadastro() {
    const input = document.getElementById('cadIdentificacao');
    const modoRadio = document.querySelector('input[name="modoTrabalho"]:checked');

    if (!input || input.value.trim().length < 3) {
        alert("Por favor, digite um nome válido.");
        return;
    }

    const ident = input.value.trim();
    const identUpper = ident.toUpperCase();
    const modo = modoRadio ? modoRadio.value : "sozinho";

    // FILTRO DE PALAVRAS E NOMES PROIBIDOS
    const palavrasProibidas = ['CUZÃO', 'CUZAO', 'DESCONHECIDO', 'ESTRANHO', 'QUALQUER', 'TURMA', 'GALERA', 'DIGÃO', 'DIGAO'];
    for (let p of palavrasProibidas) {
        if (identUpper.includes(p)) {
            alert("⚠️ Nome não permitido. Por favor, insira um nome válido e respeitoso.");
            return;
        }
    }

    if (modo === 'sozinho') {
        const regexBR = /^[A-Za-zÀ-ÖØ-öø-ÿ\s]+$/;
        if (!regexBR.test(ident) || ident.length < 3) {
            alert("⚠️ No modo 'Sozinho', insira um nome válido brasileiro (apenas letras, sem números ou caracteres especiais).");
            return;
        }
    } else if (ident.length < 4) {
        alert("⚠️ A identificação da Equipe precisa de pelo menos 4 caracteres.");
        return;
    }

    // Salvando os dados
    localStorage.setItem("usuarioIdentificacao", identUpper);
    localStorage.setItem("modoTrabalho", modo);

    // Escondendo o modal (troque pelo ID correto se necessário)
    const modalCadastro = document.getElementById('trotroModalOverlay');
    if (modalCadastro) modalCadastro.style.display = 'none';

    alert("Cadastro realizado com sucesso!");

    // Chamando as funções de inicialização
    if (typeof solicitarPermissaoNotificacao === 'function') solicitarPermissaoNotificacao();
    if (typeof solicitarPermissaoCamera === 'function') solicitarPermissaoCamera();
    
    verificarConfiguracoesIniciais();
    
    if (typeof atualizarLocalizacao === 'function') atualizarLocalizacao();
    
    setTimeout(() => {
        if (typeof clienteMQTT !== 'undefined' && clienteMQTT && clienteMQTT.connected) {
            clienteMQTT.publish(TEMA_MQTT, JSON.stringify({ acao: "PRESENCA_ONLINE", remetente: typeof meuID !== 'undefined' ? meuID : 'desconhecido', autor: getAutorFormatado() }), { qos: 0 });
        }
    }, 1500);
    
    if (typeof dispararAtualizacao === 'function') dispararAtualizacao();

    location.reload(); // Recarrega para aplicar as mudanças no sistema
}

        function carregarChatLocal() {
            const salvos = localStorage.getItem('inspecao_chat_historico');
            if (salvos) chatMessages = JSON.parse(salvos);
            const agora = Date.now();
            const umDiaMs = 86400000;
            chatMessages = chatMessages.filter(m => (agora - m.timestamp) < umDiaMs);
            localStorage.setItem('inspecao_chat_historico', JSON.stringify(chatMessages));
        }

        function renderizarChat() {
            const container = document.getElementById('listaMensagensChat');
            container.innerHTML = '';
            if (chatMessages.length === 0) {
                container.innerHTML = '<div style="text-align:center; padding:20px; color:#888; font-size:12px;">Sem mensagens recentes. Seja o primeiro a dizer olá!</div>';
                return;
            }

            chatMessages.forEach(m => {
                if (m.destino && m.destino !== 'todos') { if (m.destino !== meuID && m.remetenteId !== meuID) return; }
                const isMinha = m.remetenteId === meuID && !m.isSistema; const isSistema = m.isSistema;
                const div = document.createElement('div'); 
                div.className = `chat-msg ${isMinha ? 'minha' : 'outros'}`;
                
                if (isSistema) { div.style.background = '#FFF3E0'; div.style.border = '2px solid #FF9800'; div.style.alignSelf = 'center'; div.style.width = '90%'; div.style.textAlign = 'center'; div.style.margin = '10px 0'; }
                
                const dataFormatada = new Date(m.timestamp).toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'});
                const ehPrivada = (m.destino && m.destino !== 'todos' && !isSistema);
                const avisoPrivado = ehPrivada ? `<span style="color: #FF5252; font-size: 10px; font-weight: bold;">[PRIVADO]</span> ` : '';
                div.innerHTML = `<div class="autor" style="${isSistema ? 'color:#FF9800; font-size:13px;' : ''}">${avisoPrivado}${isMinha ? 'Você' : m.autor}</div><div class="texto" style="${isSistema ? 'font-weight:bold;' : ''}">${m.texto}</div><div class="hora">${dataFormatada}</div>`;
                container.appendChild(div);
            });
            container.scrollTop = container.scrollHeight;
        }

        function capturarFotoPoda(tipo) {
            const canvasId = tipo === 'marcar' ? 'canvasFotoMarcar' : 'canvasFotoRetirar';
            const dataHoraId = tipo === 'marcar' ? 'dataHoraFotoMarcar' : 'dataHoraFotoRetirar';
            const hiddenId = tipo === 'marcar' ? 'fotoMarcarBase64' : 'fotoRetirarBase64';
            const canvas = document.getElementById(canvasId);
            const dataHoraDiv = document.getElementById(dataHoraId);
            const hiddenInput = document.getElementById(hiddenId);

            if (!navigator.mediaDevices || !navigator.mediaDevices.getUserMedia) {
                alert("Câmera não disponível neste dispositivo ou navegador."); return;
            }

            navigator.mediaDevices.getUserMedia({ video: { facingMode: 'environment' }, audio: false })
                .then(stream => {
                    const video = document.createElement('video');
                    video.srcObject = stream; video.autoplay = true; video.playsInline = true;
                    video.style.cssText = 'position:fixed;top:0;left:0;width:100%;height:100%;z-index:99999;object-fit:cover;background:#000;';
                    const btnSnap = document.createElement('button');
                    btnSnap.textContent = '📸 Tirar Foto';
                    btnSnap.style.cssText = 'position:fixed;bottom:40px;left:50%;transform:translateX(-50%);z-index:100000;padding:16px 40px;background:#FF9800;color:white;border:none;border-radius:30px;font-size:18px;font-weight:bold;cursor:pointer;box-shadow:0 4px 15px rgba(0,0,0,0.5);';
                    const btnCancelar = document.createElement('button');
                    btnCancelar.textContent = '✖ Cancelar';
                    btnCancelar.style.cssText = 'position:fixed;top:20px;right:20px;z-index:100000;padding:10px 20px;background:rgba(0,0,0,0.7);color:white;border:none;border-radius:20px;font-size:14px;font-weight:bold;cursor:pointer;';
                    document.body.appendChild(video); document.body.appendChild(btnSnap); document.body.appendChild(btnCancelar);

                    function fecharCamera() {
                        stream.getTracks().forEach(t => t.stop());
                        video.remove(); btnSnap.remove(); btnCancelar.remove();
                    }
                    btnCancelar.onclick = fecharCamera;
                    btnSnap.onclick = () => {
                        const agora = new Date();
                        const dataStr = agora.toLocaleDateString('pt-BR') + ' ' + agora.toLocaleTimeString('pt-BR');
                        canvas.width = video.videoWidth || 640; canvas.height = video.videoHeight || 480;
                        const ctx = canvas.getContext('2d');
                        ctx.drawImage(video, 0, 0, canvas.width, canvas.height);
                        ctx.fillStyle = 'rgba(0,0,0,0.6)';
                        ctx.fillRect(0, canvas.height - 28, canvas.width, 28);
                        ctx.fillStyle = '#FFEB3B'; ctx.font = 'bold 13px Segoe UI'; ctx.textAlign = 'center';
                        ctx.fillText('📅 ' + dataStr, canvas.width / 2, canvas.height - 10);
                        hiddenInput.value = canvas.toDataURL('image/jpeg', 0.7);
                        canvas.style.display = 'block';
                        dataHoraDiv.textContent = '📅 Foto registrada em: ' + dataStr; dataHoraDiv.style.display = 'block';
                        fecharCamera();

                        const detectarCidadePorGPS = (lng, lat, callbackCidade) => {
                            fetch(`https://nominatim.openstreetmap.org/reverse?format=json&lat=${lat}&lon=${lng}&zoom=10&addressdetails=1`)
                                .then(r => r.json())
                                .then(data => {
                                    const cidade = data.address?.city || data.address?.town || data.address?.village || data.address?.municipality || data.address?.county || '';
                                    callbackCidade(cidade);
                                })
                                .catch(() => callbackCidade(''));
                        };

                        const resolverGPS = (callback) => {
                            if (userMarker) {
                                const ll = userMarker.getLngLat();
                                callback(ll.lng, ll.lat);
                            } else if (navigator.geolocation) {
                                navigator.geolocation.getCurrentPosition(
                                    pos => callback(pos.coords.longitude, pos.coords.latitude),
                                    () => callback(null, null)
                                );
                            } else { callback(null, null); }
                        };

                        if (tipo === 'marcar') {
                            const campoCidade = document.getElementById('podaCidade');
                            campoCidade.value = '🔍 Detectando cidade...';
                            campoCidade.readOnly = true;
                            campoCidade.disabled = false;
                            campoCidade.style.opacity = '0.7';
                            campoCidade.style.cursor = 'not-allowed';
                            campoCidade.style.background = '';
                            document.getElementById('confirmCidadeBox').style.display = 'none';
                            resolverGPS((lng, lat) => {
                                if (!lng) {
                                    campoCidade.value = ''; campoCidade.readOnly = false;
                                    campoCidade.style.opacity = '1'; campoCidade.style.cursor = 'text';
                                    campoCidade.placeholder = 'Cidade não detectada — digite manualmente';
                                    return;
                                }
                                detectarCidadePorGPS(lng, lat, cidade => {
                                    if (cidade) {
                                        campoCidade.value = cidade;
                                        campoCidade.style.opacity = '1';
                                        campoCidade.style.background = '#e8f5e9';
                                        dataHoraDiv.textContent += ' | 📍 ' + cidade;
                                    } else {
                                        campoCidade.value = ''; campoCidade.readOnly = false;
                                        campoCidade.style.opacity = '1'; campoCidade.style.cursor = 'text';
                                        campoCidade.placeholder = 'Cidade não detectada — digite manualmente';
                                    }
                                });
                            });
                        } else if (tipo === 'retirar') {
                            resolverGPS((lng, lat) => {
                                if (!lng) return;
                                document.getElementById('retirarGpsLat').value = lat;
                                document.getElementById('retirarGpsLng').value = lng;
                                detectarCidadePorGPS(lng, lat, cidade => {
                                    document.getElementById('retirarCidade').value = cidade;
                                    if (cidade) dataHoraDiv.textContent += ' | 📍 ' + cidade;
                                });
                            });
                        }
                    };
                })
                .catch(err => { alert("Não foi possível acessar a câmera. Verifique as permissões.\n" + err.message); });
        }

        function enviarNotificacaoSistema(titulo, corpo) {
            if ("Notification" in window && Notification.permission === "granted") {
                if (navigator.serviceWorker) {
                    navigator.serviceWorker.getRegistration().then(function(reg) {
                        if (reg && reg.showNotification) {
                            reg.showNotification(titulo, { body: corpo, icon: './icon.png', badge: './icon.png', vibrate: [200, 100, 200], requireInteraction: false });
                        } else { dispararNotificacaoSimples(titulo, corpo); }
                    });
                } else { dispararNotificacaoSimples(titulo, corpo); }
            }
        }

        function enviarNotificacaoSistemaForcada(titulo, corpo, tag) {
            if (!("Notification" in window)) return;
            const opts = { body: corpo, icon: './icon.png', badge: './icon.png', vibrate: [300, 100, 300, 100, 300], requireInteraction: true, tag: tag || 'sistema', renotify: true };
            if (Notification.permission === "granted") {
                if (navigator.serviceWorker) {
                    navigator.serviceWorker.getRegistration().then(function(reg) {
                        if (reg && reg.showNotification) { reg.showNotification(titulo, opts); }
                        else { try { new Notification(titulo, opts); } catch(e) {} }
                    });
                } else { try { new Notification(titulo, opts); } catch(e) {} }
            } else if (Notification.permission !== "denied") {
                Notification.requestPermission().then(p => {
                    if (p === "granted") enviarNotificacaoSistemaForcada(titulo, corpo, tag);
                });
            }
        }

        const meuID = Math.random().toString(36).substring(2, 10);

        function conectarSincronizacao() {

        }

        function processarMensagemRede(msg) {
            const querVisibilidade = localStorage.getItem('prefVisibilidade') !== 'false';
            const querChat = localStorage.getItem('prefChat') !== 'false';

            if (msg.acao === "ATUALIZAR_GPS") {
                if (msg.remetente === meuID || !querVisibilidade) return;
                atualizarMarcadorColega(msg.remetente, msg.autor, msg.lng, msg.lat);
            } else if (msg.acao === "PRESENCA_ONLINE") {
                if (msg.remetente === meuID) return;
                if (!colegasNoMapa[msg.remetente]) colegasNoMapa[msg.remetente] = { nome: msg.autor, lastUpdate: Date.now() };
                else { colegasNoMapa[msg.remetente].nome = msg.autor; colegasNoMapa[msg.remetente].lastUpdate = Date.now(); }
                atualizarListaChatPrivado();
                mostrarNotificacaoRealTime(`👋 <b>${msg.autor}</b> entrou no sistema.`, "#9C27B0", false);
            }
          else if (msg.acao === "NOVA_MENSAGEM_CHAT") {
                if (msg.mensagem.remetenteId === meuID) return;
                // Ignora se for mensagem privada para outra pessoa
                if (msg.mensagem.destino && msg.mensagem.destino !== 'todos' && msg.mensagem.destino !== meuID) return; 

                // 1. DETERMINA SE É UMA MENSAGEM DE SISTEMA (Poda, Chave, etc)
                const ehMensagemSistema = msg.mensagem.isSistema === true;

                // 2. FILTRO DE ATIVAÇÃO: Se chat OFF e NÃO for sistema, ignora.
                // Se for sistema, ele CONTINUA o processamento mesmo com chat OFF.
                if (!querChat && !ehMensagemSistema) return;

                // 3. SALVA NO HISTÓRICO (Sempre salva se for sistema ou se o chat estiver ON)
                chatMessages.push(msg.mensagem); 
                localStorage.setItem('inspecao_chat_historico', JSON.stringify(chatMessages)); 
                
                const panelAberto = document.getElementById('panelChat').style.display === 'flex'; 
                const appEmSegundoPlano = document.hidden;

                if (ehMensagemSistema) {
                    // --- LÓGICA PARA AVISOS DE SISTEMA (CHAVES E PODAS) ---
                    // Estes avisos SEMPRE aparecem na tela (Top Toast) em laranja, mesmo com chat OFF
                    if (!appEmSegundoPlano) {
                        mostrarNotificacaoRealTime(`🤖 <b>AVISO DO SISTEMA</b><br>${msg.mensagem.texto}`, "#FF9800", true);
                    }
                    // Se o painel estiver aberto, atualiza a lista visual
                    if (panelAberto) renderizarChat();
                    
                } else {
                    // --- LÓGICA PARA MENSAGENS NORMAIS DE CONVERSA ---
                    // Só toca som e mostra aviso se o usuário optou por usar o chat
                    if (querChat) {
                        tocarAlerta();
                        if (panelAberto) { 
                            renderizarChat(); 
                        } else {
                            const avisoP = (msg.mensagem.destino && msg.mensagem.destino !== 'todos') ? "[PRIVADO] " : "";
                            if (!appEmSegundoPlano) {
                                mostrarNotificacaoRealTime(`💬 <b>Nova Mensagem de ${msg.mensagem.autor}</b><br><span style="color:#FF5252">${avisoP}</span>"${msg.mensagem.texto}"<br><small style="color:#FFEB3B; font-weight:900;">(Toque para abrir)</small>`, "#00BCD4", true);
                            }
                            enviarNotificacaoSistema(`💬 Nova Mensagem de ${msg.mensagem.autor}`, msg.mensagem.texto);
                        }
                    }
                }
            }
            else if (msg.acao === "NOVA_PODA_V2" || msg.acao === "NOVA_PODA") {
                if (msg.remetente === meuID) return;
                if (msg.features && msg.features.length > 0) {
                    const jaExiste = geoJsonGlobal.features.some(f => f.properties.id === msg.features[0].properties.id);
                    if (!jaExiste) {
                        geoJsonGlobal.features.push(...msg.features);
                        renderizarMapa(); salvarTudo(); tocarAlerta(); 
                        if (document.getElementById('panelListaPoda') && document.getElementById('panelListaPoda').style.display === 'flex') renderizarListaPoda();
                        if (!document.hidden) mostrarNotificacaoRealTime(`🚨 <b>NOVA PODA!</b><br>${msg.autor} marcou o trecho:<br><span style="color:#FFEB3B; font-size:16px;">${msg.trecho}</span>`, "#FF9800", false);
                        enviarNotificacaoSistemaForcada("🚨 Nova Poda na Área", `${msg.autor} marcou o trecho: ${msg.trecho}`, 'poda-nova');
                    }
                } 
            } 
            else if (msg.acao === "RESOLVER_PODA") {
                if (msg.remetente === meuID) return;
                const podasRemovidas = geoJsonGlobal.features.filter(f => f.properties.id === msg.podaId);
                if (podasRemovidas.length > 0) {
                    geoJsonGlobal.features = geoJsonGlobal.features.filter(f => f.properties.id !== msg.podaId);
                    renderizarMapa(); salvarTudo();
                    if (document.getElementById('panelListaPoda') && document.getElementById('panelListaPoda').style.display === 'flex') renderizarListaPoda();
                    if (!document.hidden) mostrarNotificacaoRealTime(`✅ <b>PODA RESOLVIDA!</b><br>O utilizador <b style="color:#FFEB3B; font-size:16px;">${msg.autor}</b> concluiu o serviço no trecho:<br><span style="color:#4CAF50;">${podasRemovidas[0].properties.trechoPoda}</span>`, "#4CAF50", false);
                    enviarNotificacaoSistemaForcada("✅ Poda Resolvida", `${msg.autor} resolveu o trecho: ${podasRemovidas[0].properties.trechoPoda}`, 'poda-resolvida');
                }
            }
            else if (msg.acao === "NOVA_CHAVE" || msg.acao === "EDITAR_CHAVE") {
                if (msg.remetente === meuID) return;

                if (msg.acao === "NOVA_CHAVE") {
                    const jaExiste = geoJsonGlobal.features.some(f => f.properties.id === msg.feature.properties.id);
                    if (!jaExiste) {
                        geoJsonGlobal.features.push(msg.feature);
                    }
                } else if (msg.acao === "EDITAR_CHAVE") {
                    let index = geoJsonGlobal.features.findIndex(f => f.properties.id === msg.feature.properties.id);
                    if (index === -1 && msg.feature.geometry && msg.feature.geometry.coordinates) {
                        const mc = msg.feature.geometry.coordinates;
                        index = geoJsonGlobal.features.findIndex(f => {
                            if (!f.geometry || !f.geometry.coordinates) return false;
                            const fc = f.geometry.coordinates;
                            return JSON.stringify(fc) === JSON.stringify(mc);
                        });
                    }
                    if (index !== -1) {
                        geoJsonGlobal.features[index] = { ...geoJsonGlobal.features[index], properties: { ...geoJsonGlobal.features[index].properties, ...msg.feature.properties } };
                    } else {
                        geoJsonGlobal.features.push(msg.feature);
                    }
                }

                renderizarMapa(); 
                salvarTudo();
                
                if (!document.hidden) {
                    mostrarNotificacaoRealTime(msg.msgChat, "#E040FB", false);
                }
                enviarNotificacaoSistemaForcada("⚡ Atualização de Rede", msg.msgChat.replace(/<[^>]*>?/gm, ''), 'rede-chave');
            }
            else if (msg.acao === "REMOVER_CHAVE") {
                if (msg.remetente === meuID) return;
                const idx = geoJsonGlobal.features.findIndex(f => {
                    if (!f.properties.isPoint) return false;
                    const nome = (f.properties.name || "").toUpperCase().trim();
                    const n = (msg.nome || "").toUpperCase();
                    return nome === n || nome.split(/[\s,;/\-]+/).includes(n);
                });
                if (idx !== -1) { geoJsonGlobal.features.splice(idx, 1); renderizarMapa(); salvarTudo(); }
                if (!document.hidden) mostrarNotificacaoRealTime(msg.msgChat, "#F44336", false);
            }
            else if (msg.acao === "FICOU_INVISIVEL") {
                const idColega = msg.remetente;
                if (colegasNoMapa[idColega]) { colegasNoMapa[idColega].marker.remove(); delete colegasNoMapa[idColega]; }
            }
            else if (msg.acao === "NOVA_INTERLIGACAO") {
                if (msg.remetente === meuID) return;
                const jaExiste = geoJsonGlobal.features.some(f => f.properties.id === msg.feature.properties.id);
                if (!jaExiste) {
                    geoJsonGlobal.features.push(msg.feature);
                    (msg.featuresPontos||[]).forEach(fp => {
                        const TOL = 0.0005;
                        const idx = geoJsonGlobal.features.findIndex(f => f.properties.isPoint && f.geometry?.coordinates && Math.abs(f.geometry.coordinates[0]-fp.geometry.coordinates[0])<TOL && Math.abs(f.geometry.coordinates[1]-fp.geometry.coordinates[1])<TOL);
                        if (idx !== -1) { geoJsonGlobal.features[idx].properties.corPonto = fp.properties.corPonto; geoJsonGlobal.features[idx].properties.statusTxt = fp.properties.statusTxt; }
                        else geoJsonGlobal.features.push(fp);
                    });
                    renderizarMapa(); salvarTudo();
                    if (!document.hidden) mostrarNotificacaoRealTime(`🔗 <b>NOVA INTERLIGAÇÃO!</b><br>${msg.msgChat ? msg.msgChat.replace(/<[^>]*>?/gm,' ') : msg.feature.properties.name}`, '#FF5722', false);
                    if (typeof enviarNotificacaoSistemaForcada === 'function') enviarNotificacaoSistemaForcada("🔗 Nova Interligação de Rede", (msg.autor||'') + ": " + msg.feature.properties.name, 'interligacao');
                }
            }
            else if (msg.acao === "REMOVER_INTERLIGACAO") {
                if (msg.remetente === meuID) return;
                const antes = geoJsonGlobal.features.length;
                geoJsonGlobal.features = geoJsonGlobal.features.filter(f => f.properties.id !== msg.featId);
                if (geoJsonGlobal.features.length !== antes) { renderizarMapa(); salvarTudo(); }
                if (!document.hidden) mostrarNotificacaoRealTime(`🗑️ Interligação removida por ${msg.autor||'colega'}.`, '#F44336', false);
            }
            else if (msg.acao === "PING_SCAN") {
                if (msg.remetente !== meuID && estouVisivel && userMarker) {
                    const lngLat = userMarker.getLngLat();
                    clienteMQTT.publish(TEMA_MQTT, JSON.stringify({ acao: "PONG_SCAN", remetente: meuID, destinatario: msg.remetente, autor: getAutorFormatado(), lng: lngLat.lng, lat: lngLat.lat }), { qos: 0 });
                }
            }
            else if (msg.acao === "PONG_SCAN") {
                if (msg.destinatario === meuID) {
                    atualizarMarcadorColega(msg.remetente, msg.autor, msg.lng, msg.lat);
                    mostrarNotificacaoRealTime(`📡 ${msg.autor} respondeu — está online no mapa!`, "#4CAF50", false);
                }
            }
        }

        let _cidadeDetectadaTemp = '';
        async function tentarDetectarCidadePoda() {
            const de = document.getElementById('podaDe').value.trim().toUpperCase();
            const ate = document.getElementById('podaAte').value.trim().toUpperCase();
            if (!de || !ate) return;
            if (document.getElementById('podaCidade').value) return; // já tem cidade

            const ptDe = buscarChaveExata(de);
            const ptAte = buscarChaveExata(ate);
            if (!ptDe && !ptAte) return;

            const coords = (ptDe || ptAte).geometry.coordinates;
            const lng = coords[0]; const lat = coords[1];
            try {
                const resp = await fetch(`https://nominatim.openstreetmap.org/reverse?format=json&lat=${lat}&lon=${lng}&zoom=10&addressdetails=1`);
                const data = await resp.json();
                const cidade = data.address?.city || data.address?.town || data.address?.village || data.address?.municipality || data.address?.county;
                if (cidade) {
                    _cidadeDetectadaTemp = cidade;
                    document.getElementById('cidadeDetectadaNome').textContent = cidade;
                    document.getElementById('confirmCidadeBox').style.display = 'block';
                }
            } catch(e) { recusarCidadeDetectada(); }
        }

        function removerPodaManualmente() {
            const equipe = document.getElementById('retirarEquipe').value.trim();
            const de = document.getElementById('retirarDe').value.trim().toUpperCase();
            const ate = document.getElementById('retirarAte').value.trim().toUpperCase();
            const fotoBase64 = document.getElementById('fotoRetirarBase64').value;
            const cidadeRetirada = document.getElementById('retirarCidade').value.trim();
            const retirarLat = parseFloat(document.getElementById('retirarGpsLat').value);
            const retirarLng = parseFloat(document.getElementById('retirarGpsLng').value);

            if (!equipe || !de || !ate) { alert("Por favor, preencha o número da equipe e os IDs das duas chaves (De / Até)."); return; }
            if (!fotoBase64) { alert("📸 Foto obrigatória! Tire uma foto do local antes de retirar a pendência."); return; }

            const podasAtivas = geoJsonGlobal.features.filter(f => f.properties.name === "🚨 PODA PENDENTE" || f.properties.name === "🌳 PODA DE ÁRVORE");
            let idsParaRemover = []; let trechoReal = ""; let podaRef = null;
            for (let p of podasAtivas) {
                if (p.properties.trechoPoda) {
                    const trechoUp = p.properties.trechoPoda.toUpperCase();
                    if (trechoUp.includes(de) && trechoUp.includes(ate)) {
                        idsParaRemover.push(p.properties.id);
                        trechoReal = p.properties.trechoPoda;
                        if (!podaRef) podaRef = p.properties;
                    }
                }
            }
            if (idsParaRemover.length === 0) { alert(`❌ Nenhuma poda pendente encontrada entre as chaves ${de} e ${ate}.\nVerifique se digitou os IDs corretamente.`); return; }

            // Verificação de localização: GPS da retirada vs GPS da marcação
            if (podaRef && podaRef.gpsMarcarLat && podaRef.gpsMarcarLng && !isNaN(retirarLat) && !isNaN(retirarLng)) {
                const distancia = calcularDistanciaMetros(podaRef.gpsMarcarLat, podaRef.gpsMarcarLng, retirarLat, retirarLng);
                const LIMITE_METROS = 500;
                if (distancia > LIMITE_METROS) {
                    const distFormatada = distancia >= 1000
                        ? (distancia / 1000).toFixed(1) + ' km'
                        : Math.round(distancia) + ' metros';
                    alert(
                        `⚠️ LOCAIS DIFERENTES!\n\n` +
                        `Você está a ${distFormatada} de distância do local onde a poda foi marcada.\n\n` +
                        `A retirada de pendência só pode ser feita no mesmo local onde a poda foi registrada.\n\n` +
                        `Aproxime-se do local correto e tente novamente.`
                    );
                    return;
                }
            }

            geoJsonGlobal.features = geoJsonGlobal.features.filter(f => !idsParaRemover.includes(f.properties.id));
            renderizarMapa(); salvarTudo();
            if (document.getElementById('panelListaPoda') && document.getElementById('panelListaPoda').style.display === 'flex') renderizarListaPoda();

            const nomeRemetente = equipe.toLowerCase().includes("equipe") ? equipe : `Equipe ${equipe}`;
            const obsRetirar = (document.getElementById('retirarObservacao').value || "").trim();
            const dataHora = new Date().toLocaleString('pt-BR');
            const cidadeStr = cidadeRetirada ? ` | 📍 <b>${cidadeRetirada}</b>` : '';
            const obsStr = obsRetirar ? `<br>📝 <i>${obsRetirar}</i>` : '';
            enviarMensagemSistemaChat(`✅ <b>PODA RESOLVIDA</b>${cidadeStr}<br>A <b>${nomeRemetente}</b> retirou a pendência nas chaves <b style="color:#4CAF50;">${trechoReal}</b>!<br>🕐 ${dataHora}${obsStr}`);
            enviarNotificacaoSistemaForcada("✅ Poda Resolvida", `${nomeRemetente} retirou pendência em ${trechoReal}${cidadeRetirada ? ' (' + cidadeRetirada + ')' : ''}`, 'poda-retirada');
            if (clienteMQTT && clienteMQTT.connected) {
                idsParaRemover.forEach(idEncontrado => {
                    clienteMQTT.publish(TEMA_MQTT, JSON.stringify({ acao: "RESOLVER_PODA", remetente: meuID, autor: nomeRemetente, podaId: idEncontrado, cidade: cidadeRetirada, dataHora: dataHora, fotoRetirar: fotoBase64 }), { qos: 0 });
                });
            }
            document.getElementById('retirarPodaModalOverlay').style.display = 'none';
            alert(`✅ Pendência retirada com sucesso!\nO trecho ${trechoReal} está limpo e todo o grupo foi avisado no chat.`);
        }

        let podasAvisadas = {}; let podaAtualParaResolver = null;
        function verificarProximidadePodas(lng, lat) {
            if (document.getElementById('resolvePodaModalOverlay').style.display === 'flex') return;
            const podas = geoJsonGlobal.features.filter(f => f.properties.name === "🚨 PODA PENDENTE" || f.properties.name === "🌳 PODA DE ÁRVORE");
            const ptUser = turf.point([lng, lat]);

            for (let poda of podas) {
                const id = poda.properties.id;
                if (podasAvisadas[id]) continue; 
                const ptPoda = turf.point(poda.geometry.coordinates);
                const distMeters = turf.distance(ptUser, ptPoda, { units: 'meters' });
                if (distMeters <= 50) {
                    podaAtualParaResolver = id;
                    document.getElementById('podaTrechoAviso').innerText = poda.properties.trechoPoda || "Desconhecido";
                    document.getElementById('resolvePodaModalOverlay').style.display = 'flex';
                    podasAvisadas[id] = true;
                    enviarNotificacaoSistema("⚠️ RADAR DE PODA", `Está a menos de 50 metros do trecho: ${poda.properties.trechoPoda}`);
                    break;
                }
            }
        }

        function resolverPoda(resolvida) {
            document.getElementById('resolvePodaModalOverlay').style.display = 'none';
            if (resolvida && podaAtualParaResolver) {
                const autorDaAcao = getAutorFormatado();
                const podaRef = geoJsonGlobal.features.find(f => f.properties.id === podaAtualParaResolver);
                const trecho = podaRef ? podaRef.properties.trechoPoda : "Desconhecido";
                geoJsonGlobal.features = geoJsonGlobal.features.filter(f => f.properties.id !== podaAtualParaResolver);
                renderizarMapa(); salvarTudo();
                if (document.getElementById('panelListaPoda') && document.getElementById('panelListaPoda').style.display === 'flex') renderizarListaPoda();
                enviarMensagemSistemaChat(`✅ <b>PODA RESOLVIDA</b><br>A pendência do trecho <b style="color:#4CAF50;">${trecho}</b> foi removida pela equipa/utilizador <b>${autorDaAcao}</b>!`);
                if (clienteMQTT && clienteMQTT.connected) {
                    const payload = { acao: "RESOLVER_PODA", remetente: meuID, autor: autorDaAcao, podaId: podaAtualParaResolver };
                    clienteMQTT.publish(TEMA_MQTT, JSON.stringify(payload));
                }
                alert("✅ Pendência de poda resolvida! Os X vermelhos sumiram do mapa e a equipa foi notificada no CHAT.");
            }
            podaAtualParaResolver = null;
        }

        function calcularDistanciaMetros(lat1, lng1, lat2, lng2) {
            const R = 6371000;
            const dLat = (lat2 - lat1) * Math.PI / 180;
            const dLng = (lng2 - lng1) * Math.PI / 180;
            const a = Math.sin(dLat/2) * Math.sin(dLat/2) +
                      Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) *
                      Math.sin(dLng/2) * Math.sin(dLng/2);
            return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
        }

        function processarKML(kmlStr, fileName) {
            const parser = new DOMParser();
            const xml = parser.parseFromString(kmlStr, "text/xml");
            const gj = toGeoJSON.kml(xml);
            if (!gj || !gj.features) return;

            const novosNomes = [...new Set(
                gj.features
                    .filter(f => f.geometry && f.geometry.type && !f.geometry.type.includes('Point'))
                    .map(f => ((f.properties && f.properties.name) || "S/N").toUpperCase())
            )];
            novosNomes.forEach(n => { if(!circuitosState[n]) circuitosState[n] = { visible: true, color: paleta[Object.keys(circuitosState).length % paleta.length] }; });

            gj.features.forEach(f => {
                try {
                    if (!f.geometry || !f.geometry.type) return;
                    if (!f.properties) f.properties = {};
                    let feat = { type: 'Feature', geometry: f.geometry, properties: { ...f.properties } };
                    if (!feat.properties.id) feat.properties.id = 'kml_' + Date.now() + '_' + Math.random().toString(36).substr(2, 9);
                    feat.properties.sourceFile = fileName;
                    let nAtivo = (feat.properties.name || "S/N").toUpperCase();
                    let descLimpa = (feat.properties.description || "").replace(/<[^>]*>?/gm, ' ').toUpperCase();
                    let fullTxt = nAtivo + " " + descLimpa;
                    let vinculo = novosNomes.find(c => fullTxt.includes(c)) || "OUTROS";
                    feat.properties.alimRef = vinculo;
                    if (!f.geometry.type.includes('Point')) {
                        feat.properties.isPoint = false; feat.properties.corRede = circuitosState[nAtivo] ? circuitosState[nAtivo].color : '#4285F4'; feat.properties.alimRef = nAtivo;
                    } else {
                        feat.properties.isPoint = true;
                        if (/\bNA\b/.test(fullTxt)) { feat.properties.corPonto = '#00FF00'; feat.properties.statusTxt = "Aberta (NA)"; }
                        else if (/\bNF\b/.test(fullTxt)) { feat.properties.corPonto = '#FF0000'; feat.properties.statusTxt = "Fechada (NF)"; }
                        else { feat.properties.corPonto = '#888'; feat.properties.statusTxt = "Equipamento"; }
                    }
                    geoJsonGlobal.features.push(feat);
                } catch(e) { console.warn("Feature ignorada:", e); }
            });
            renderizarMapa(); atualizarPainel();
            if(document.getElementById('panelArquivos').style.display === 'flex') atualizarPainelArquivos();
            salvarTudo();
        }

function criarIconeTriangulo(corBase, tamanho) {
    const canvas = document.createElement('canvas');
    canvas.width = tamanho;
    canvas.height = tamanho;
    const ctx = canvas.getContext('2d');
    
    // Desenha o fundo do triângulo (amarelo)
    ctx.fillStyle = corBase;
    ctx.beginPath();
    ctx.moveTo(tamanho / 2, 0); // Vértice do topo (centralizado)
    ctx.lineTo(0, tamanho);      // Vértice inferior esquerdo
    ctx.lineTo(tamanho, tamanho);// Vértice inferior direito
    ctx.closePath();
    ctx.fill();
    
    // Desenha a borda preta (para dar destaque)
    ctx.strokeStyle = '#000000';
    ctx.lineWidth = 2;
    ctx.stroke();
    
    return ctx.getImageData(0, 0, tamanho, tamanho);
}

function criarIconeQuadrado(corBase, tamanho) {
    const canvas = document.createElement('canvas');
    canvas.width = tamanho;
    canvas.height = tamanho;
    const ctx = canvas.getContext('2d');
    
    // Desenha o fundo do quadrado
    ctx.fillStyle = corBase;
    ctx.fillRect(0, 0, tamanho, tamanho);
    
    // Desenha a borda preta (para dar destaque no mapa)
    ctx.strokeStyle = '#000000';
    ctx.lineWidth = 2;
    ctx.strokeRect(1, 1, tamanho - 2, tamanho - 2);
    
    return ctx.getImageData(0, 0, tamanho, tamanho);
}

function aplicarFiltros() {
    // 1. Filtro dos Circuitos (Linhas) - Metodologia 100% segura
    const v = Object.keys(circuitosState).filter(k => circuitosState[k].visible);
    v.push("OUTROS");
    
    const filterAlim = ['any'];
    v.forEach(circ => filterAlim.push(['==', ['coalesce', ['get', 'alimRef'], 'OUTROS'], circ]));

    // Atalho blindado para ler o nome da chave (se for nulo, devolve vazio)
    const getNome = ['coalesce', ['get', 'name'], ''];

    // 2. Filtro dos Pontos
    const filterPontos = ['any'];

    if (filtrosStatus.verdes) {
        filterPontos.push(
            ['==', ['coalesce', ['get', 'corPonto'], ''], '#00FF00'],
            ['>=', ['index-of', 'NA', getNome], 0],
            ['>=', ['index-of', 'RA', getNome], 0]
        );
    }

if (filtrosStatus.vermelhos) {
        filterPontos.push(
            ['==', ['coalesce', ['get', 'corPonto'], ''], '#FF0000'],
            ['>=', ['index-of', 'NF', getNome], 0],
            ['>=', ['index-of', 'RC', getNome], 0],
            
            // 🟢 CORREÇÃO: O botão vermelho só mantém FA, FD e FU na tela se a cor deles NÃO for verde (#00FF00)
            ['all', ['>=', ['index-of', 'FA', getNome], 0], ['!=', ['coalesce', ['get', 'corPonto'], ''], '#00FF00']],
            ['all', ['>=', ['index-of', 'FD', getNome], 0], ['!=', ['coalesce', ['get', 'corPonto'], ''], '#00FF00']],
            ['all', ['>=', ['index-of', 'FU', getNome], 0], ['!=', ['coalesce', ['get', 'corPonto'], ''], '#00FF00']],
            ['all', ['>=', ['index-of', 'RE', getNome], 0], ['!=', ['coalesce', ['get', 'corPonto'], ''], '#00FF00']]
  );
}
    if (filtrosStatus.laranjas) {
        filterPontos.push(
            ['==', ['coalesce', ['get', 'corPonto'], ''], '#FF9800']
        );
    }
    if (filtrosStatus.cinzas) {
        filterPontos.push(
            ['==', ['coalesce', ['get', 'corPonto'], ''], '#888'],
            ['==', ['coalesce', ['get', 'corPonto'], ''], '#808080'], 
            ['==', ['coalesce', ['get', 'tipoEquip'], ''], 'TRAFO'],
            ['==', ['coalesce', ['get', 'tipoEquip'], ''], 'REGULADOR'],
            ['==', ['coalesce', ['get', 'tipoEquip'], ''], 'CAPACITOR'],
            ['>=', ['index-of', 'KVA', ['upcase', getNome]], 0]
        );
    }

    if (filterPontos.length === 1) {
        filterPontos.push(false); 
    }

    // Impede que as siglas gerem círculos no fundo
    const filtroNaoESigla = ['all',
        ['==', ['index-of', 'NA', ['upcase', getNome]], -1],
        ['==', ['index-of', 'NF', getNome], -1],
        ['==', ['index-of', 'FA', getNome], -1],
        ['==', ['index-of', 'FD', getNome], -1],
        ['==', ['index-of', 'FU', getNome], -1],
        ['==', ['index-of', 'RA', getNome], -1],
        ['==', ['index-of', 'RC', getNome], -1],
        ['==', ['index-of', 'KVA', ['upcase', getNome]], -1]
]

    // ==========================================
    // 3. APLICAÇÃO SEGURA NAS CAMADAS
    // ==========================================

    if(map.getLayer('l-linhas')) {
        map.setFilter('l-linhas', ['all', filterAlim, ['==', ['get', 'isPoint'], false], ['!=', ['geometry-type'], 'Polygon']]);
    }
    if(map.getLayer('l-linhas-contorno')) {
        map.setFilter('l-linhas-contorno', ['all', filterAlim, ['==', ['get', 'isPoint'], false], ['!=', ['geometry-type'], 'Polygon']]);
    }
    if(map.getLayer('l-poligonos-fill')) {
        map.setFilter('l-poligonos-fill', ['all', filterAlim, ['==', ['get', 'isPoint'], false], ['==', ['geometry-type'], 'Polygon']]);
    }

    if(map.getLayer('l-labels')) {
        map.setFilter('l-labels', [
            'all', 
            filterAlim, 
            ['==', ['get', 'isPoint'], true], 
            filterPontos, 
            ['!=', getNome, '🌳 PODA DE ÁRVORE'], 
            ['!=', getNome, '🚨 PODA PENDENTE']
        ]);
    }
    
// ... (restante do código aplicarFiltros) ...

    if(map.getLayer('l-pontos-quadrados')) {
        map.setFilter('l-pontos-quadrados', [
            'all',
            filterAlim,
            filterPontos,
            ['!=', ['get', 'tipoEquip'], 'TRAFO'], // 🚫 REGRA DE OURO: Proíbe o Trafo de ser quadrado!
            ['any', 
                ['==', ['coalesce', ['get', 'corPonto'], ''], '#00FF00'], 
                ['==', ['coalesce', ['get', 'corPonto'], ''], '#FF9800'],
                ['==', ['coalesce', ['get', 'corPonto'], ''], '#FF0000'],
                ['>=', ['index-of', 'FA', getNome], 0],
                ['>=', ['index-of', 'FD', getNome], 0],
                ['>=', ['index-of', 'FU', getNome], 0]
            ]
        ]);
    }

    if(map.getLayer('l-pontos-triangulos-kva')) {
        map.setFilter('l-pontos-triangulos-kva', [
            'all',
            filterAlim,
            filterPontos,
            ['==', ['get', 'isPoint'], true],
            // 🟡 REGRA DE OURO: Puxa o Trafo para a camada de Triângulos!
            ['any', 
                ['>=', ['index-of', 'KVA', ['upcase', getNome]], 0],
                ['==', ['get', 'tipoEquip'], 'TRAFO'] 
            ]
        ]);
    }
    if(map.getLayer('l-pontos-circulos')) {
        map.setFilter('l-pontos-circulos', [
            'all',
            filterAlim,
            filterPontos,
            ['==', ['get', 'isPoint'], true],
            ['!=', ['coalesce', ['get', 'corPonto'], ''], '#FF9800'],
            filtroNaoESigla,
            // 🟢 REGRA ESPECÍFICA NO FILTRO DINÂMICO
            ['!', ['all', ['match', ['get', 'tipoEquip'], ['RE', 'FU', 'FD', 'FA'], true, false], ['==', ['coalesce', ['get', 'corPonto'], ''], '#00FF00']]]
        ]);
    }

    if(map.getLayer('l-podas-x-grande')) {
        map.setFilter('l-podas-x-grande', ['==', getNome, '🚨 PODA PENDENTE']);
    }
    if(map.getLayer('l-podas-x')) {
        map.setFilter('l-podas-x', ['all', filterAlim, ['==', getNome, '🌳 PODA DE ÁRVORE'], filterPontos]);
    }
};

function salvarEdicaoChavePorNumero() {
    const numAntigo = document.getElementById('chaveNumeroAntigo').value.trim().toUpperCase();
    const numNovo = document.getElementById('chaveNumeroNovo').value.trim().toUpperCase();
    
    const statusSelect = document.getElementById('chaveStatusEdicaoNum');
    const statusValue = statusSelect.value.toUpperCase().trim();
    // Pega o texto EXATO que aparece na lista (com emojis e tudo)
    const statusTextoCombo = statusSelect.options[statusSelect.selectedIndex].text.toUpperCase();
    
    const kva = statusValue === 'TRAFO' ? (document.getElementById('chaveKvaEdicao').value || '').trim() : '';
    
    if (!numAntigo || !numNovo) { alert("Preencha o número atual e o novo número da chave."); return; }
    if (statusValue === 'TRAFO' && !kva) { alert("Por favor, informe a potência em KVA."); return; }

    const index = geoJsonGlobal.features.findIndex(f => {
        if (!f.properties.isPoint) return false;
        const nome = (f.properties.name || "").toUpperCase().trim();
        return nome === numAntigo || nome.split(/[\s,;/\-]+/).includes(numAntigo);
    });
    
    if (index === -1) { alert(`❌ Chave "${numAntigo}" não encontrada no mapa.`); return; }

    // 1. ACHAR A SIGLA EM QUALQUER LUGAR DA PALAVRA (ex: 1010-RE)
    const matchSigla = numNovo.match(/(RE|FU|FD|FA|NA|NF|RA|RC)/i);
    const siglaExtraida = matchSigla ? matchSigla[0].toUpperCase() : statusValue;

// 2. DEDUZIR A COR (À Prova de Falhas)
    let cor = '#888888'; // Cinza padrão para Trafos e outros

    if (statusValue === 'NA') {
        cor = '#00FF00'; // Verde
    } else if (statusValue === 'NF' || statusValue === 'FU' || statusValue === 'FD' || statusValue === 'FA') {
        cor = '#FF0000'; // Vermelho
    } else if (statusValue === 'RE') {
        // Resolve o mistério do RE Aberto ou Fechado!
        if (statusTextoCombo.includes('🟢') || statusTextoCombo.includes('ABERTO') || statusTextoCombo.includes('NA')) {
            cor = '#00FF00';
        } else {
            cor = '#FF0000';
        }
    } else if (statusValue === 'TRAFO' || statusValue === 'TRANSFORMADOR') {
        cor = '#888888'; // 🟢 Trafo volta a ser Cinza (Garante que não vire quadrado Laranja)
    } else {
        const corFallback = corParaStatus(statusValue);
        if (corFallback) cor = corFallback.toUpperCase();
    }

    const statusLabel = labelParaStatus(statusValue, kva);
    const alimRef = geoJsonGlobal.features[index].properties.alimRef || "N/A";

    // 3. GRAVA TODAS AS PROPRIEDADES NO MAPA
    geoJsonGlobal.features[index] = {
        ...geoJsonGlobal.features[index],
        properties: { 
            ...geoJsonGlobal.features[index].properties, 
            name: numNovo, 
            statusTxt: statusLabel, 
            corPonto: cor,          // Vai ficar #00FF00 ou #FF0000 certinho
            tipoEquip: statusValue, 
            kva: kva,
            status: statusValue,    
            sigla: siglaExtraida    // Agora salva a sigla corretamente
        }
    };

    const featureAtualizada = geoJsonGlobal.features[index];
    const autor = getAutorFormatado();
    const tipoDesc = statusValue === 'TRAFO' ? `TRAFO ${kva}KVA` : statusValue;
    const msgChat = `✏️ <b>CHAVE ATUALIZADA</b><br>👤 <b>${autor}</b> | 🔌 Circuito: <b style="color:#9C27B0;">${alimRef}</b><br><b style="color:#FF9800;">${numAntigo}</b> → <b style="color:#4CAF50;">${numNovo}</b> | Tipo: <b>${tipoDesc}</b>`;

    renderizarMapa(); 
    salvarTudo();

        function validarFormatoChave(input) {
            let v = input.value.toUpperCase();
            v = v.replace(/[^0-9A-Z\-]/g, '');
            const partes = v.split('-');
            const nums = partes[0].replace(/\D/g, '').slice(0, 7);
            if (partes.length === 1) {
                v = nums;
            } else {
                const sufixo = partes.slice(1).join('').replace(/[^A-Z0-9]/g, '').slice(0, 7);
                v = nums + (sufixo ? '-' + sufixo : '-');
            }
            if (input.value !== v) input.value = v;
        }

       function desenharLinhaRegua() {
            if (ptsM.length < 2) return;
            let features = []; let totalDist = 0;
            for (let i = 0; i < ptsM.length - 1; i++) {
                const distSeg = turf.distance(ptsM[i], ptsM[i+1], {units: 'meters'});
                totalDist += distSeg;
                let distSegTxt = distSeg >= 1000 ? (distSeg/1000).toLocaleString('pt-BR', {minimumFractionDigits: 2, maximumFractionDigits: 2}) + ' km' : distSeg.toLocaleString('pt-BR', {minimumFractionDigits: 1, maximumFractionDigits: 1}) + ' m';
                features.push({ type: 'Feature', geometry: { type: 'LineString', coordinates: [ptsM[i], ptsM[i+1]] }, properties: { distancia: distSegTxt } });
            }
            let totalTxt = totalDist >= 1000 ? (totalDist/1000).toLocaleString('pt-BR', {minimumFractionDigits: 2, maximumFractionDigits: 2}) + ' km' : totalDist.toLocaleString('pt-BR', {minimumFractionDigits: 1, maximumFractionDigits: 1}) + ' m';
            if (ptsM.length > 2) { features.push({ type: 'Feature', geometry: { type: 'Point', coordinates: ptsM[ptsM.length - 1] }, properties: { distancia: 'Total: ' + totalTxt } }); }
            map.getSource('src-regua-percurso').setData({ type: 'FeatureCollection', features: features });
            if (!popupMedicao) popupMedicao = new maplibregl.Popup({closeOnClick: false, closeButton: false});
            popupMedicao.setLngLat(ptsM[ptsM.length - 1]).setHTML(`<div style="font-size:16px; font-weight:900; color:#FF5252; padding:5px;">📏 Total: ${totalTxt}</div>`).addTo(map);
        }

        function getDistMetros(lng1, lat1, lng2, lat2) {
            let dx = lng1 - lng2; let dy = lat1 - lat2; let cosLat = Math.cos(lat1 * Math.PI / 180);
            return Math.sqrt(Math.pow(dx * 111320 * cosLat, 2) + Math.pow(dy * 111320, 2));
        }
        
        function getDistanceToSegment(px, py, ax, ay, bx, by) {
            let l2 = Math.pow(ax - bx, 2) + Math.pow(ay - by, 2);
            if (l2 === 0) return getDistMetros(px, py, ax, ay);
            let t = ((px - ax) * (bx - ax) + (py - ay) * (by - ay)) / l2;
            t = Math.max(0, Math.min(1, t));
            let projX = ax + t * (bx - ax); let projY = ay + t * (by - ay); return getDistMetros(px, py, projX, projY);
        }
        
        function extractSegments(geom) {
            let segs = [];
            if(!geom) return segs;
            function addLine(coords) { for(let i=0; i<coords.length-1; i++){ if(coords[i] && coords[i+1]) segs.push([coords[i], coords[i+1]]); } }
            if(geom.type === 'LineString') addLine(geom.coordinates);
            else if(geom.type === 'MultiLineString') geom.coordinates.forEach(addLine);
            else if(geom.type === 'GeometryCollection') geom.geometries.forEach(g => { if(g.type === 'LineString') addLine(g.coordinates); else if(g.type === 'MultiLineString') g.coordinates.forEach(addLine); });
            return segs;
        }

window.calcularCorrenteChave = function(uid, tipoChave) {
    const inputClientesElem = document.getElementById('inputClientes_' + uid);
    if (!inputClientesElem) return;

    const qtdClientes = parseInt(inputClientesElem.value) || 0;
    if (qtdClientes === 0) {
        alert("⚡ Por favor, digite a quantidade de clientes que passam por esta chave.");
        return;
    }

    const hora = new Date().getHours();
    let horarioAtual, demandaAtualVA;

    if (hora >= 0 && hora < 6) { 
        horarioAtual = 'Madrugada (00h–06h) Carga Leve'; demandaAtualVA = 200; 
    } else if (hora >= 6 && hora < 18) { 
        horarioAtual = 'Dia (06h–18h) Carga Média'; demandaAtualVA = 600; 
    } else { 
        horarioAtual = 'Noite (18h–00h) Horário de Pico'; demandaAtualVA = 1500; 
    }

    const selectMT = document.getElementById('selMt_' + uid);
    const tensaoLinha = selectMT ? parseInt(selectMT.value) : 13800;
    const raizDeTres = 1.732;

    let capacidadeNominal = null;
    let textoCapacidade = "";
    let eloSugeridoAlerta = "";

    if (tipoChave === 'FU') {
        const inputElo = document.getElementById('inputElo_' + uid);
        const eloInput = inputElo ? inputElo.value.toUpperCase().trim() : "";
        const matchElo = eloInput.match(/(\d+)/);
        
        if (matchElo) {
            capacidadeNominal = parseFloat(matchElo[1]);
            textoCapacidade = `Capacidade do Elo: <b>${eloInput}</b> (Suporta ${capacidadeNominal}A contínuos)`;
        } else {
            textoCapacidade = `<span style="color:#F44336">⚠️ Informe o Elo (ex: 25K) para ver o limite.</span>`;
        }
    } else if (tipoChave === 'DI') {
        const inputCapDI = document.getElementById('inputCapacidadeDI_' + uid);
        let capDI = inputCapDI ? parseInt(inputCapDI.value) : NaN;
        
        if (isNaN(capDI) || capDI <= 0) {
             capDI = (qtdClientes <= 3500) ? 800 : 1200;
             textoCapacidade = `Disjuntor (Inferido): <b>${capDI}A</b>`;
             capacidadeNominal = capDI;
        } else {
             capacidadeNominal = capDI;
             textoCapacidade = `Disjuntor Ajustado para: <b>${capacidadeNominal}A</b>`;
        }
    } else {
        capacidadeNominal = (tipoChave === 'FA') ? 800 : ((tipoChave === 'FD') ? 600 : 400); 
        textoCapacidade = `Corrente Nominal da Chave: <b>${capacidadeNominal}A</b> <em style="font-size:10px;">(${tipoChave === 'FA' ? 'Faca Direta' : 'Lâmina'})</em>`;
    }

    const gerarBlocoSimulacao = (demandaVA, tituloHorario, isPico) => {
        const potEstimadaVA = qtdClientes * demandaVA;
        const potEstimadaKVA = potEstimadaVA / 1000;
        const correnteEst = potEstimadaVA / (raizDeTres * tensaoLinha);
        
        let alertaCor = '#2196F3', alertaTxt = '✅ CÁLCULO REALIZADO', pct = 0;

        if (capacidadeNominal !== null && capacidadeNominal > 0) {
            pct = (correnteEst / capacidadeNominal) * 100;
            if (pct > 100) { 
                alertaCor = '#F44336'; 
                alertaTxt = tipoChave === 'FU' ? '🔴 RISCO DE QUEIMA DO ELO!' : '🔴 RISCO DE DESARME!'; 
                if (tipoChave === 'FU' && isPico) {
                     const eloRecomendado = sugerirEloIdeal(correnteEst);
                     eloSugeridoAlerta = `<div style="margin-top: 6px; background-color: #fff3cd; padding: 6px; border-left: 4px solid #ffc107; border-radius: 4px; font-size:10px;"><span style="color: #856404; font-weight: bold;">⚠️ Elo Subdimensionado para o Pico!</span><br><strong>💡 Sugestão de Troca:</strong> ${eloRecomendado}</div>`;
                }
            } else if (pct > 80) { 
                alertaCor = '#FF9800'; 
                alertaTxt = '⚠️ ATENÇÃO: Próximo do limite'; 
            } else { 
                alertaCor = '#4CAF50'; 
                alertaTxt = '✅ CORRENTE PASSANTE SEGURA'; 
            }
        }

        const bgCor = isPico ? '#FFF3E0' : '#F3F9FF'; 
        const bordaCor = isPico ? '#FFB74D' : '#64B5F6';

        return `
            <div style="background:${bgCor}; border-left:4px solid ${bordaCor}; padding:8px; margin-top:8px; border-radius:4px;">
                <div style="border-bottom:1px solid #ddd; padding-bottom:4px; margin-bottom:6px;"><b>${tituloHorario}</b></div>
                <div>Potência Passante: <b>${potEstimadaKVA.toFixed(1)} kVA</b></div>
                <div style="margin-top:4px; font-size:13px;">Corrente Estimada: <b style="color:#000;">${correnteEst.toFixed(1)} A</b></div>
                ${capacidadeNominal ? `<div style="margin-top:2px; font-size:10px; color:#555;">Carregamento: ${pct.toFixed(1)}%</div>` : ''}
                <div style="margin-top:6px; color:${alertaCor}; font-weight:900; font-size:11px; background:#fff; padding:4px; border-radius:4px; text-align:center; border: 1px solid ${alertaCor};">${alertaTxt}</div>
                ${eloSugeridoAlerta}
            </div>
        `;
    };

    const tensaoKV = (tensaoLinha / 1000).toFixed(1);
    let htmlFinal = `
        <div style="font-size:10px; color:#666; border-bottom:1px solid #ddd; padding-bottom:4px; margin-bottom:4px;">Rede Primária (${tensaoKV}kV Trifásico)</div>
        <div>${textoCapacidade}</div>
    `;

    if (demandaAtualVA === 1500) {
        htmlFinal += gerarBlocoSimulacao(1500, `🕐 AGORA: ${horarioAtual}`, true);
    } else {
        htmlFinal += gerarBlocoSimulacao(demandaAtualVA, `🕐 AGORA: ${horarioAtual}`, false);
        eloSugeridoAlerta = ""; // Reseta o alerta para não duplicar no próximo bloco

window.calcularCargaTrafoPopup = function(popupId, kvaTrafo) {
    const inputElement = document.getElementById('qtdCli_' + popupId);
    if (!inputElement) return;

    const qtdClientes = parseInt(inputElement.value) || 1;
    const hora = new Date().getHours();
    
    let horarioAtual, demandaAtualVA;
    if (hora >= 0 && hora < 6) { 
        horarioAtual = 'Madrugada (00h–06h) Carga Leve'; demandaAtualVA = 200; 
    } else if (hora >= 6 && hora < 18) { 
        horarioAtual = 'Dia (06h–18h) Carga Média'; demandaAtualVA = 600; 
    } else { 
        horarioAtual = 'Noite (18h–00h) Horário de Pico'; demandaAtualVA = 1500; 
    }

    const selectBT = document.getElementById('selBt_' + popupId);
    const tensaoLinha = selectBT ? parseInt(selectBT.value) : 220;
    const raizDeTres = 1.732;
    const correnteMax = (kvaTrafo * 1000) / (raizDeTres * tensaoLinha);

    const gerarBlocoSimulacao = (demandaVA, tituloHorario, isPico) => {
        const potEstimadaVA = qtdClientes * demandaVA;
        const correnteEst = potEstimadaVA / (raizDeTres * tensaoLinha);
        const pct = (correnteEst / correnteMax) * 100;
        
        const potEstimadaKVA = potEstimadaVA / 1000;
        const saldoKVA = kvaTrafo - potEstimadaKVA; 
        
        let textoReserva = "";
        if (saldoKVA >= 0) {
            const clientesLivres = Math.floor((saldoKVA * 1000) / demandaVA);
            textoReserva = `<div style="margin-top:6px; background:#E8F5E9; color:#2E7D32; padding:4px; border-radius:4px; font-size:10px; border: 1px solid #4CAF50;">
                🟢 <b>Sobram:</b> ${saldoKVA.toFixed(1)} kVA (+${clientesLivres} casas)
            </div>`;
        } else {
            const deficit = Math.abs(saldoKVA);
            textoReserva = `<div style="margin-top:6px; background:#FFEBEE; color:#C62828; padding:4px; border-radius:4px; font-size:10px; border: 1px solid #F44336;">
                🔴 <b>Faltam:</b> ${deficit.toFixed(1)} kVA (Trafo Subdimensionado)
            </div>`;
        }

        let alertaCor, alertaTxt;
        if (pct > 100) { 
            alertaCor = '#F44336'; alertaTxt = '⚠️ ALERTA: SOBRECARGA!'; 
        } else if (pct > 85) { 
            alertaCor = '#FF9800'; alertaTxt = '⚠️ ATENÇÃO: Próximo do limite'; 
        } else { 
            alertaCor = '#4CAF50'; alertaTxt = '✅ TRAFO NORMAL'; 
        }

        const bgCor = isPico ? '#FFF3E0' : '#F3F9FF';
        const bordaCor = isPico ? '#FFB74D' : '#64B5F6';

        return `
            <div style="background:${bgCor}; border-left:4px solid ${bordaCor}; padding:8px; margin-top:8px; border-radius:4px;">
                <div style="border-bottom:1px solid #ddd; padding-bottom:4px; margin-bottom:4px;"><b>${tituloHorario}</b></div>
                <div>Corrente Estimada: <b>${correnteEst.toFixed(1)} A / fase</b></div>
                <div>Carregamento Atual: <b>${pct.toFixed(1)}%</b></div>
                ${textoReserva}
                <div style="margin-top:6px; color:${alertaCor}; font-weight:900; font-size:11px; background:#fff; padding:4px; border-radius:4px; text-align:center; border: 1px solid ${alertaCor};">${alertaTxt}</div>
            </div>
        `;
    };

    let htmlFinal = `
        <div style="font-size:10px; color:#666; border-bottom:1px solid #ddd; padding-bottom:4px; margin-bottom:4px;">Cálculo Trifásico (${tensaoLinha}V)</div>
        <div>Corrente Nominal Máx: <b>${correnteMax.toFixed(1)} A / fase</b></div>
    `;

    if (demandaAtualVA === 1500) {
        htmlFinal += gerarBlocoSimulacao(1500, `🕐 AGORA: ${horarioAtual}`, true);
    } else {
        htmlFinal += gerarBlocoSimulacao(demandaAtualVA, `🕐 AGORA: ${horarioAtual}`, false);
        htmlFinal += gerarBlocoSimulacao(1500, `🌙 PROJEÇÃO: Noite (18h–00h) Horário de Pico`, true);
    }

    const resDiv = document.getElementById('res_' + popupId);
    if(resDiv) {
        resDiv.style.display = 'block';
        resDiv.innerHTML = htmlFinal;
    }
}; // <----- ESTA ERA A CHAVE QUE ESTAVA FALTANDO E QUEBRANDO SEU CÓDIGO!

function abrirPopupPonto(e) {
    if(medindo) return;

    const ptLng = e.lngLat.lng; 
    const ptLat = e.lngLat.lat;
    
    const exC = e.features[0].geometry?.coordinates;
    const exactLng = (exC && exC.length >= 2) ? exC[0] : ptLng;
    const exactLat = (exC && exC.length >= 2) ? exC[1] : ptLat;

    let minDistsPorAlim = {}; 
    const alimAtivos = Object.keys(circuitosState);
    
    geoJsonGlobal.features.forEach(feat => {
        if (!feat.properties.isPoint && feat.properties.alimRef) {
            let nomeAlim = feat.properties.alimRef;
            if (nomeAlim === "OUTROS" || !alimAtivos.includes(nomeAlim)) return;
            let segmentos = extractSegments(feat.geometry);
            segmentos.forEach(seg => {
                let d = getDistanceToSegment(ptLng, ptLat, seg[0][0], seg[0][1], seg[1][0], seg[1][1]);
                if (!minDistsPorAlim[nomeAlim] || d < minDistsPorAlim[nomeAlim]) { minDistsPorAlim[nomeAlim] = d; }
            });
        }
    });
    
    let alimsOrdenados = Object.keys(minDistsPorAlim).map(nome => ({ nome, dist: minDistsPorAlim[nome] })).filter(a => a.dist < 1500).sort((a, b) => a.dist - b.dist);
    let alimentadorExibicao = "Não Identificado";

    if (alimsOrdenados.length > 0) {
        if (f.corPonto === '#00FF00' && alimsOrdenados.length >= 2) {
            alimentadorExibicao = `<div style="display:flex; justify-content:center; align-items:center; gap:4px; margin-top:4px; flex-wrap:wrap;"><span style="color:var(--roxo); background:#fff; padding:3px 6px; border-radius:4px; border:1px solid #ccc; font-weight:900;">${alimsOrdenados[0].nome}</span><span style="color:#FF5252; font-weight:900; font-size:16px;">⇄</span><span style="color:var(--roxo); background:#fff; padding:3px 6px; border-radius:4px; border:1px solid #ccc; font-weight:900;">${alimsOrdenados[1].nome}</span></div>`;

    let blocoManobra = '';
    const corPontoNorm = (f.corPonto || '').toUpperCase();
    if (corPontoNorm === '#00FF00') {
        const TOL_EXTREMO = 0.0005;
        const TOL_MEDIO  = 0.0003;
        const pLng2 = ptLng; const pLat2 = ptLat;

        function pontoPerto(c, tol) { return Array.isArray(c) && Math.abs(c[0] - pLng2) < tol && Math.abs(c[1] - pLat2) < tol; }
        function conectaAoCirculo(geom) {
            if (!geom) return false;
            let segs = [];
            if (geom.type === 'LineString') segs = [geom.coordinates];
            else if (geom.type === 'MultiLineString') segs = geom.coordinates;
            else if (geom.type === 'GeometryCollection') {
                (geom.geometries || []).forEach(g => {
                    if (g.type === 'LineString') segs.push(g.coordinates);
                    else if (g.type === 'MultiLineString') g.coordinates.forEach(s => segs.push(s));
                });
            }
            for (const seg of segs) {
                if (!seg || seg.length < 2) continue;
                if (pontoPerto(seg[0], TOL_EXTREMO) || pontoPerto(seg[seg.length - 1], TOL_EXTREMO)) return true;
                for (let i = 1; i < seg.length - 1; i++) { if (pontoPerto(seg[i], TOL_MEDIO)) return true; }

window.confirmarInterligacao = function() {
    const radioSel = document.querySelector('input[name="interlig_circ"]:checked');
    if (!radioSel) { alert('Selecione o circuito para a cor da linha.'); return; }

    const lng1 = Number(_il.lng1); const lat1 = Number(_il.lat1);
    const lng2 = Number(_il.lng2); const lat2 = Number(_il.lat2);

    if (isNaN(lng1) || isNaN(lat1) || isNaN(lng2) || isNaN(lat2)) { alert("Erro crítico: Coordenadas inválidas. O mapa foi protegido."); return; }
    if (lng1 === lng2 && lat1 === lat2) { alert("Atenção: A chave de origem e destino possuem a mesma coordenada."); return; }

    _il.circCor = radioSel.value;
    _il.corLinha = (typeof circuitosState !== 'undefined' && circuitosState[_il.circCor]) ? circuitosState[_il.circCor].color : '#4285F4';
    
    const autor = typeof getAutorFormatado === 'function' ? getAutorFormatado() : (window.meuNome || 'Usuário');
    const dataHora = new Date().toLocaleString('pt-BR');
    const novoId = 'interlig_' + Date.now();

    let distLabel = '';
    try { 
        if (typeof turf !== 'undefined') {
            const dk = turf.distance(turf.point([lng1, lat1]), turf.point([lng2, lat2]), {units: 'kilometers'}); 
            distLabel = dk < 1 ? (dk * 1000).toFixed(0) + ' m' : dk.toFixed(3) + ' km'; 
        }
    } catch(e) { console.warn("Aviso: Falha ao calcular distância Turf.js"); }

    const novaLinha = {
        type: 'Feature',
        geometry: { type: 'LineString', coordinates: [[lng1, lat1], [lng2, lat2]] },
        properties: { id: novoId, name: 'INTERLIG: ' + _il.nome1 + ' ⇄ ' + _il.nome2, isPoint: false, alimRef: _il.circCor, corRede: _il.corLinha, sourceFile: 'INTERLIGACAO', tipoInterlig: _il.tipo, autor: autor, dataRegisto: dataHora, distLabel: distLabel }
    };

    geoJsonGlobal.features.push(novaLinha);

    if (_il.tipo === 'paralelo' || _il.tipo === 'anel' || _il.tipo === 'nf') {
        const corManobra = _il.tipo === 'nf' ? '#FF0000' : '#00FF00';
        const stManobra = _il.tipo === 'nf' ? 'Fechada (NF)' : 'Aberta (NA)';
        const mLng = _il.pontoManobra === 'chave2' ? lng2 : lng1;
        const mLat = _il.pontoManobra === 'chave2' ? lat2 : lat1;
        const mNome = _il.pontoManobra === 'chave2' ? _il.nome2 : _il.nome1;
        
        _atualizarOuCriarCirculo(mLng, mLat, mNome, corManobra, stManobra, _il.circCor);

        if (_il.pontoManobra === 'chave2') {
            if (_il.acaoChave1 === 'apagar') {
                for (let i = geoJsonGlobal.features.length - 1; i >= 0; i--) {
                    let f = geoJsonGlobal.features[i];
                    if (f && f.geometry && f.geometry.type === 'Point' && f.geometry.coordinates) {
                        let coords = f.geometry.coordinates;
                        if (Math.abs(coords[0] - lng1) < 0.0005 && Math.abs(coords[1] - lat1) < 0.0005) { geoJsonGlobal.features.splice(i, 1); }
                    }
                }
            } else {
                const corAcao = _il.acaoChave1 === 'na' ? '#00FF00' : '#FF0000';
                const stAcao = _il.acaoChave1 === 'na' ? 'Aberta (NA)' : 'Fechada (NF)';
                _atualizarOuCriarCirculo(lng1, lat1, _il.nome1, corAcao, stAcao, _il.circ1);
            }
        }
    }

    try {
        renderizarMapa();
        if (typeof salvarTudo === 'function') salvarTudo();
        fecharModalInterligacao();
        if (typeof mostrarNotificacaoRealTime === 'function') {
            mostrarNotificacaoRealTime('🔗 <b>Interligação criada com sucesso!</b><br>'+_il.nome1+' ⇄ '+_il.nome2+(distLabel?' ('+distLabel+')':''),'#4CAF50',false);
        }
    } catch (erroMapa) { console.error("Erro", erroMapa); }
};

function _atualizarOuCriarCirculo(lng, lat, nome, corPonto, statusTxt, alimRef) {
    if (isNaN(lng) || isNaN(lat)) return; 
    const TOL = 0.0005;
    const ex = geoJsonGlobal.features.find(f => f.properties.isPoint && f.geometry?.coordinates && f.geometry.type === 'Point' && Math.abs(f.geometry.coordinates[0]-lng) < TOL && Math.abs(f.geometry.coordinates[1]-lat) < TOL);
    
    if (ex) {
        ex.properties.corPonto = corPonto;
        ex.properties.statusTxt = statusTxt;
    } else {
        geoJsonGlobal.features.push({
            type: 'Feature',
            geometry: { type: 'Point', coordinates: [Number(lng), Number(lat)] },
            properties: { id: 'pt_'+Date.now(), name: nome, isPoint: true, corPonto: corPonto, statusTxt: statusTxt, alimRef: alimRef||'OUTROS', sourceFile: 'INTERLIGACAO' }
        });
    }
}

 function toggleUI() {
    // Localizamos a div que agrupa todos os botões e o novo botão de voltagem
    const mainUI = document.getElementById('mainUI');
    const badgeRadar = document.getElementById('radarOnlineBadge');
    const avisoZona = document.getElementById('avisoZona');
    const fabEnergia = document.querySelector('.fab-container'); // CAPTURA O NOVO BOTÃO ⚡
    const fabMenu = document.getElementById('fab-menu'); // CAPTURA O MENU SELETOR

    if (mainUI) {
        if (mainUI.style.display === 'none') {
            // Se está escondido, mostra tudo de volta
            mainUI.style.display = 'block';
            if(badgeRadar) badgeRadar.style.display = 'block';
            if(avisoZona) avisoZona.style.display = 'block';
            if(fabEnergia) fabEnergia.style.display = 'flex'; // MOSTRA O BOTÃO ⚡
            
            document.getElementById('btnToggleUI').innerText = '👁️';
        } else {
            // Se está visível, esconde a interface inteira
            mainUI.style.display = 'none';
            if(badgeRadar) badgeRadar.style.display = 'none';
            if(avisoZona) avisoZona.style.display = 'none';
            
            // ESCONDE O BOTÃO ⚡ E O MENU SELEÇÃO (caso esteja aberto)
            if(fabEnergia) fabEnergia.style.display = 'none';
            if(fabMenu) fabMenu.style.display = 'none';
            
            document.getElementById('btnToggleUI').innerText = '🙈';
        }
    } else {
        console.error("Erro: A div 'mainUI' não foi encontrada no seu HTML.");
    }
}

const btnOlho = document.getElementById('btnToggleUI');
let isDragging = false;
let offsetX, offsetY;

// Eventos para Celular (Touch)
btnOlho.addEventListener('touchstart', (e) => {
    isDragging = true;
    const touch = e.touches[0];
    // Calcula a distância entre o toque e a borda do botão
    offsetX = touch.clientX - btnOlho.getBoundingClientRect().left;
    offsetY = touch.clientY - btnOlho.getBoundingClientRect().top;
    btnOlho.style.transition = "none"; // Remove animação durante o arraste
}, { passive: true });

window.addEventListener('touchmove', (e) => {
    if (!isDragging) return;
    const touch = e.touches[0];
    
    // Define a nova posição baseada no movimento do dedo
    let x = touch.clientX - offsetX;
    let y = touch.clientY - offsetY;
    
    // Impede que o botão saia da tela
    x = Math.max(0, Math.min(x, window.innerWidth - btnOlho.offsetWidth));
    y = Math.max(0, Math.min(y, window.innerHeight - btnOlho.offsetHeight));
    
    btnOlho.style.left = x + 'px';
    btnOlho.style.top = y + 'px';
    btnOlho.style.right = 'auto'; // Anula o 'right: 10px' do CSS original
}, { passive: false });

window.addEventListener('touchend', () => {
    isDragging = false;
});

// Eventos para Computador (Mouse) - Opcional
btnOlho.addEventListener('mousedown', (e) => {
    isDragging = true;
    offsetX = e.clientX - btnOlho.getBoundingClientRect().left;
    offsetY = e.clientY - btnOlho.getBoundingClientRect().top;
    btnOlho.style.transition = "none";
});

window.addEventListener('mousemove', (e) => {
    if (!isDragging) return;
    let x = e.clientX - offsetX;
    let y = e.clientY - offsetY;
    btnOlho.style.left = x + 'px';
    btnOlho.style.top = y + 'px';
    btnOlho.style.right = 'auto';
});

window.addEventListener('mouseup', () => {
    isDragging = false;
});

    // Dá meio segundo para a tela estabilizar e faz a verificação
    setTimeout(() => {
        verificarCadastroPendente();
    }, 500);
});

function verificarCadastroPendente() {
    // Procura na memória se o usuário já tem uma identificação salva
    const ident = localStorage.getItem("usuarioIdentificacao");
    
    // Se NÃO existir identificação, é a primeira vez (ou foi resetado)
    if (!ident) {
        // ATENÇÃO: Substitua 'ID_DO_MODAL_DE_CADASTRO' pelo ID real da sua div de registro
        // Exemplo: 'configModalOverlay' ou 'modalCadastro'
        const modal = document.getElementById('ID_DO_MODAL_DE_CADASTRO');
        
        if (modal) {
            modal.style.display = 'flex';
        }
    } else {
        // Se o usuário JÁ ESTIVER CADASTRADO, chama a tela de boas vindas normal
        verificarAvisoBoasVindas();
    }
}
